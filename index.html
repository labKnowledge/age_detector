from PIL import Image from io import BytesIO from transformers import ViTFeatureExtractor, ViTForImageClassification import gradio as gr def genPrediction(im): # Init model, transforms model = ViTForImageClassification.from_pretrained('nateraw/vit-age-classifier') transforms = ViTFeatureExtractor.from_pretrained('nateraw/vit-age-classifier') # Transform our image and pass it through the model inputs = transforms(im, return_tensors='pt') output = model(**inputs) # Predicted Class probabilities proba = output.logits.softmax(1) # Predicted Classes return proba.argmax(1) # pipe = await pipeline("image-classification", model="nateraw/vit-age-classifier") app = gr.Interface(fn=genPrediction, inputs=gr.Image(), outputs=["text"]) app.launch() transformers-js-py
